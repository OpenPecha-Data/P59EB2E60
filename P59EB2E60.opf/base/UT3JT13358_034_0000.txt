༄༅། །པོད་ཆེན་དྲུག་གི་རྩ་བའི་དཀར་ཆག། ༠༠༡ དགེ་སློང་ཕའི་སོ་སོར་ཐར་པའི་མདོ། ༠༠༢ འདུལ་བ་མདོ་རྩ་བ། ༠༠༣ འཕགས་པ་གཞི་ཐམས་ཅད་ཡོད་པར་སྨྲ་བའི་དགེ་ཚུལ་གྱི་ཚིག་ལེའུར་བྱས་པ། ཀཱ་རི་ཀཱ་ལྔ་བཅུ། ༠༠༤ འཕགས་པ་གཞི་ཐམས་ཅད་ཡོད་པར་སྨྲ་བའི་དགེ་ཚུལ་གྱི་ཚིག་ལེའུར་བྱས་པ། ཀཱ་རི་ཀཱ་སུམ་ཅུ། ༠༠༥ ཆོས་མངོན་པའི་མཛོད་ཀྱི་ཚིག་ལེའུར་བྱས་པ། ༠༠༦ ཆོས་མངོན་པ་ཀུན་ལས་བཏུས་པ། ༠༠༧ དབུ་མ་རྩ་བའི་ཚིག་ལེའུར་བྱས་པ་ཤེས་རབ་ཅེས་བྱ་བ། ༠༠༨ དབུ་མ་ལ་འཇུག་པ་ཞེས་བྱ་བ། ༠༠༩ ཞིབ་མོ་རྣམ་པར་འཐག་པ་ཞེས་བྱ་བའི་མདོ། ༠༡༠ རིགས་པ་དྲུག་ཅུ་པའི་ཚིག་ལེའུར་བྱས་པ། ༠༡༡ རྩོད་པ་བཟློག་པའི་ཚིག་ལེའུར་བྱས་པ། ༠༡༢ སྟོང་པ་ཉིད་བདུན་ཅུ་པའི་ཚིག་ལེའུར་བྱས་པ། ༠༡༣ བསྟན་བཅོས་བཞི་བརྒྱ་པ་ཞེས་བྱ་བའི་ཚིག་ལེའུར་བྱས་པ། ༠༡༤ བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པ་ལ་འཇུག་པ། ༠༡༥ ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་མན་ངག་གི་བསྟན་བཅོས་མངོན་པར་རྟོགས་པའི་རྒྱན། ༠༡༦ ཐེག་པ་ཆེན་པོ་མདོ་སྡེའི་རྒྱན་ཞེས་བྱ་བའི་ཚིག་ལེའུར་བྱས་པ། ༠༡༧ དབུས་དང་མཐའ་རྣམ་པར་འབྱེད་པའི་ཚིག་ལེའུར་བྱས་པ། ༠༡༨ ཆོས་དང་ཆོས་ཉིད་རྣམ་པར་འབྱེད་པ། ༠༡༩ ཆོས་དང་ཆོས་ཉིད་རྣམ་པར་འབྱེད་པའི་ཚིག་ལེའུར་བྱས་པ། ༠༢༠ ཐེག་པ་ཆེན་པོ་རྒྱུད་བླ་མའི་བསྟན་བཅོས། ༠༢༡ བཤེས་པའི་སྤྲིང་ཡིག ༠༢༢ དབུ་མ་རྒྱན་གྱི་ཚིག་ལེའུར་བྱས་པ། ༠༢༣ ཞིབ་མོ་རྣམ་པར་འཐག་པ་ཞེས་བྱ་བའི་རབ་ཏུ་བྱེད་པ། ༠༢༤ སྟོང་པ་ཉིད་བདུན་ཅུ་པའི་འགྲེལ་པ། ༠༢༥ རྩོད་པ་བཟློག་པའི་འགྲེལ་པ། ༠༢༦ རིགས་པ་དྲུག་ཅུ་པའི་འགྲེལ་པ། ༠༢༧ ཕྱོགས་ཐམས་ཅད་ལས་རྣམ་པར་རྒྱལ་བ་ཆེན་པོ་འཇམ་མགོན་ས་སྐྱ་པཎྜི་ཏ་ཀུན་དགའ་རྒྱལ་མཚན་དཔལ་བཟང་པོའི་ཞབས་ཀྱི་གསུང་རབ་གླེགས་བམ་གཉིས་པ་ལས་ཚད་མ་རིགས་པའི་གཏེར་ཞེས་བྱ་བའི་བསྟན་བཅོས། ༠༢༨ སྡོམ་པ་གསུམ་གྱི་རབ་ཏུ་དབྱེ་བ་ཞེས་བྱ་བ། ༠༢༩ སྡོམ་པ་གསུམ་གྱི་རབ་ཏུ་དབྱེ་བའི་ཁ་སྐོང་གཞི་ལམ་འབྲས་གསུམ་གསལ་བར་བྱེད་པའི་ལེགས་བཤད་འོད་ཀྱི་སྣང་བ་ཞེས་བྱ་བ།
